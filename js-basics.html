三、表达式

四、流程控制

// --- if / else if / else ---
let temperature = 28;
if (temperature >= 35) {
  console.log("高温预警");
} else if (temperature >= 25) {
  console.log("天气较热：" + temperature + "°C");
} else {
  console.log("气温适宜");
}

// --- switch ---
let day = new Date().getDay();  // 0=周日, 6=周六
switch (day) {
  case 0: case 6:
    console.log("今天是周末"); break;
  default:
    console.log("今天是工作日（" + day + "）");
}

// --- for 循环 ---
let forResult = "";
for (let i = 1; i <= 5; i++) {
  forResult += i + " ";
}
console.log("for 1-5:", forResult.trim());

// --- while 循环 ---
let n = 1, whileResult = 0;
while (n <= 100) {
  whileResult += n;
  n++;
}
console.log("while 1+2+...+100 =", whileResult);

// --- for...of 遍历数组 ---
let fruits = ["苹果", "香蕉", "橙子"];
let fruitStr = "";
for (let fruit of fruits) {
  fruitStr += fruit + " ";
}
console.log("for...of:", fruitStr.trim());

// --- for...in 遍历对象属性 ---
let person = { name: "王五", age: 22, city: "北京" };
let personStr = "";
for (let key in person) {
  personStr += key + ":" + person[key] + "  ";
}
console.log("for...in:", personStr.trim());

// --- break 与 continue ---
let breakDemo = "";
for (let i = 0; i < 10; i++) {
  if (i === 3) continue;   // 跳过 3
  if (i === 7) break;      // 遇到 7 退出
  breakDemo += i + " ";
}
console.log("break/continue:", breakDemo.trim());

五、函数

// 1. 函数声明（可提升）
function add(a, b) {
  return a + b;
}
console.log("add(3, 4) =", add(3, 4));

// 2. 函数表达式（赋值给变量）
const multiply = function(a, b) {
  return a * b;
};
console.log("multiply(6, 7) =", multiply(6, 7));

// 3. 箭头函数（ES6，简洁语法）
const square = n => n * n;
const greet  = (name, title = "同学") => `你好，${title}${name}！`;
console.log("square(9) =", square(9));
console.log(greet("小明"));
console.log(greet("老师", ""));

// 4. 默认参数 & 剩余参数
function sum(...nums) {
  return nums.reduce((acc, cur) => acc + cur, 0);
}
console.log("sum(1,2,3,4,5) =", sum(1, 2, 3, 4, 5));

// 5. 递归函数（计算阶乘）
function factorial(n) {
  if (n <= 1) return 1;
  return n * factorial(n - 1);
}
console.log("factorial(6) =", factorial(6));

// 6. 高阶函数（函数作为参数）
const nums = [1, 2, 3, 4, 5, 6];
const evens = nums.filter(n => n % 2 === 0);
const doubled = nums.map(n => n * 2);
console.log("filter 偶数:", evens.join(", "));
console.log("map 翻倍:", doubled.join(", "));

// 7. 闭包（函数访问外部作用域变量）
function makeCounter(start = 0) {
  let count = start;
  return {
    increment() { return ++count; },
    decrement() { return --count; },
    value()     { return count;   }
  };
}
const counter = makeCounter(10);
counter.increment(); counter.increment(); counter.decrement();
console.log("闭包计数器值:", counter.value()); // 11
